Ordenação Rápida

Técnicas de Programação

QuickSort

QUICKSORT(A, INI, FIM)

IF INI >= FIM - 1 THEN
    RETURN
END

LP = PARTICIONA(A, INI, FIM)
QUICKSORT(A, INI, LP)
QUICKSORT(A, LP+1, FIM)

MERGE-SORT

MERGE-SORT(A, AUX, INI, FIM)

IF INI >= FIM - 1 THEN
    RETURN
END

MEIO <- (INI + FIM - 1) / 2
MERGE-SORT(A, AUX, INI, MEIO+1)
MERGE-SORT(A, AUX, MEIO+1, FIM)

MERGE(A, AUX, INI, FIM, MEIO+1)
